# FlowPane 布局

FlowPane 是流动布局，窗口发生变化组件会自动流动，如果窗口变宽则子节点会尽量显示在一行，不能一行显示时会在下一行显示。

// ========================= 布局例子 =========================
FlowPane pane = new FlowPane(); // 定义流布局
pane.setPadding(new Insets(8)); // 设置内边距
pane.setHgap(5); //设置控件之间的水平间隔距离
pane.setVgap(5); //设置控件之间的垂直间隔距离

Label lbName = new Label("请输入用户名：");
TextField tfName = new TextField();
Label lbPassword = new Label("请输入密码：");
TextField tfPassword = new TextField();
Button okbtn = new Button("Ok");

// 把组件加入 pane
pane.getChildren().addAll(lbName, tfName, lbPassword, tfPassword, okbtn);

Scene flowScene = new Scene(pane);
stage.setScene(flowScene);

stage.show(); // 主窗口显示

# GridPane 布局

GridPane 是网格布局，下面是使用方法，我们可以在 GridPane 中创建很多行和列，我们也可以控制控件放到那行那列。

// ========================= 布局例子 =========================
GridPane gridPane = new GridPane();
gridPane.setAlignment(Pos.CENTER); // 设置对齐
gridPane.setPadding(new Insets(11));
gridPane.setHgap(5);
gridPane.setVgap(5);

// gridPane.add(控件, 列, 行); 行/列都是从 0 开始
gridPane.add(new Button("1"), 0, 0);
gridPane.add(new Button("2"), 1, 0);
gridPane.add(new Button("3"), 0, 1);
gridPane.add(new Button("4"), 1, 1);
gridPane.add(new Button("5"), 0, 2);
gridPane.add(new Button("6"), 1, 2);

Scene gridPaneScene = new Scene(gridPane);
stage.setScene(gridPaneScene);

stage.show(); // 主窗口显示

# BorderPane 布局

BorderPane 布局会把整个 Scene 划分为五个区域，分别为：上，中，下，左，右。

BorderPane borderPane = new BorderPane();
// 顶部 flowpane 布局
FlowPane flowPaneTop = new FlowPane();
flowPaneTop.setPadding(new Insets(10));
flowPaneTop.setAlignment(Pos.CENTER);
Label labelTop = new Label("TOP");
flowPaneTop.getChildren().add(labelTop);

// 底部 flowpane 布局
FlowPane flowPaneBottom = new FlowPane();
flowPaneBottom.setPadding(new Insets(10));
flowPaneBottom.setAlignment(Pos.CENTER);
Label labelBottom = new Label("BOTTOM");
flowPaneBottom.getChildren().add(labelBottom);

// 左侧 flowpane 布局
FlowPane flowPaneLeft = new FlowPane();
flowPaneLeft.setPadding(new Insets(10));
flowPaneLeft.setAlignment(Pos.CENTER);
Label labelLeft = new Label("LEFT");
flowPaneLeft.getChildren().add(labelLeft);

// 右侧 flowpane 布局
FlowPane flowPaneRight = new FlowPane();
flowPaneRight.setPadding(new Insets(10));
flowPaneRight.setAlignment(Pos.CENTER);
Label labelRight = new Label("RIGHT");
flowPaneRight.getChildren().add(labelRight);

// 中间 flowpane 布局
FlowPane flowPaneCenter = new FlowPane();
flowPaneCenter.setPadding(new Insets(10));
flowPaneCenter.setAlignment(Pos.CENTER);
Label labelCenter = new Label("CENTER");
flowPaneCenter.getChildren().add(labelCenter);


borderPane.setTop(flowPaneTop);
borderPane.setBottom(flowPaneBottom);
borderPane.setLeft(flowPaneLeft);
borderPane.setRight(flowPaneRight);
borderPane.setCenter(flowPaneCenter);
borderPane.setMinSize(350, 200);



Scene gridPaneScene = new Scene(borderPane);
stage.setScene(gridPaneScene);

stage.show(); // 主窗口显示

# StackPane 布局

StackPane 布局，是一种堆叠布局，可以把一些控件堆起来布局。

// ========================= 布局例子 =========================
// 绘制一个圆形
Circle circle = new Circle(300, 135, 100);
circle.setFill(Color.DARKSLATEBLUE);
circle.setStroke(Color.BLACK);

// 绘制一个球体
Sphere sphere = new Sphere(50);

// 创建一个文本
Text text = new Text("Hello,JavaFX!");

text.setFont(Font.font(null, FontWeight.BOLD, 115));

text.setFill(Color.CRIMSON);

text.setX(20);
text.setY(50);

// 创建一个 StackPane 布局
StackPane stackPane = new StackPane();
ObservableList list = stackPane.getChildren();

list.addAll(circle, sphere, text);

Scene stackPaneScene = new Scene(stackPane);
stage.setScene(stackPaneScene);

stage.show(); // 主窗口显示

# TextFlow 文本流

// ========================= 布局例子 =========================
        Text text = new Text("Hello,JavaFX!");
        text.setFont(new Font(15));
        text.setFill(Color.DARKSLATEBLUE);

        Text text1 = new Text("Hello,JavaFX!");
        text1.setFont(new Font(15));
        text1.setFill(Color.YELLOWGREEN);

        Text text2 = new Text("Hello,JavaFX!");
        text2.setFont(new Font(15));
        text2.setFill(Color.GREEN);

        TextFlow textFlow = new TextFlow();
        textFlow.setPadding(new Insets(15));
        textFlow.setTextAlignment(TextAlignment.JUSTIFY);
        textFlow.setPrefSize(600, 300);

        // 设置行间距
        textFlow.setLineSpacing(5.0);

        ObservableList list = textFlow.getChildren();
        list.addAll(text1, text2, text);
        
        Scene stackPaneScene = new Scene(textFlow);
        stage.setScene(stackPaneScene);

        stage.show(); // 主窗口显示

# TilePane

TilePane 是一个类似于 FlowPane 的容器。 但是它与 FlowPane 不同，因为子组件列在相同大小的单元格上。

// ========================= 布局例子 =========================
        Button btn1 = new Button("Button1");
        Button btn2 = new Button("Button2");
        Button btn3 = new Button("Button3");
        Button btn4 = new Button("Button4");
        Button btn5 = new Button("Button5");
        Button btn6 = new Button("Button6");

        TilePane tilePane = new TilePane();
        tilePane.setPadding(new Insets(8));
        tilePane.setHgap(5);
        tilePane.setVgap(5);
        tilePane.getChildren().add(btn1);
        tilePane.getChildren().add(btn2);
        tilePane.getChildren().add(btn3);
        tilePane.getChildren().add(btn4);
        tilePane.getChildren().add(btn5);
        tilePane.getChildren().add(btn6);
        tilePane.setTileAlignment(Pos.TOP_LEFT);

        Scene tilePaneScene = new Scene(tilePane);
        stage.setScene(tilePaneScene);

        stage.show(); // 主窗口显示
